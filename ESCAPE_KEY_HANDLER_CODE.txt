## 1. Add handleKeyDown function

**Location:** After `handleClick()` function (around line 440)

```typescript
/**
 * Handle keydown for escaping selection mode
 */
function handleKeyDown(e: KeyboardEvent): void {
  if (!selectionMode) return;

  if (e.key === 'Escape') {
    e.preventDefault();
    e.stopPropagation();

    // Notify parent that user cancelled selector
    window.parent.postMessage({
      type: 'exit-selection-mode'
    }, parentOrigin);

    // Deactivate selection mode locally
    disableSelectionMode();
  }
}
```
