```typescript
      !settingsChanged(session.currentSettings, options);
    
    if (canReuse) {
      // Hot-swap model/permissions on active generator
      if (options.model !== session.currentModel) {
        await session.query.setModel(mappedModel);
      }
      // Push new prompt into existing generator queue
      session.sendMessage(prompt);
    } else {
      // Create new SDK session with fresh generator
      setSession(sessionId, newSession);
      this.processWithGenerator(sessionId, prompt, options);
    }
  }
}
```

```typescript
private async processWithGenerator(
  sessionId: string,
  initialPrompt: string,
  options: QueryOptions
) {
  // 1. Create message queue for multi-turn conversation
  const promptQueue = new AsyncQueue<string>([initialPrompt]);
  const session = getSession(sessionId);
  session.sendMessage = (msg) => promptQueue.push(msg);
  
  // 2. Build SDK options (environment, checkpoint resume, etc.)
  const sdkOptions = buildSdkOptions(sessionId, envForClaude, options);
  
  // 3. Start SDK session with resumption support
  if (!options.resume) {
    // Auto-resume from previous agent_session_id in DB
    const savedId = lookupAgentSessionId(sessionId);
    if (savedId) options.resume = savedId;
  }
  
  const promptInput = buildPromptIterable(promptQueue, sessionId);
  const queryResult = claudeSDK({ prompt: promptInput, options: sdkOptions });
  setQuery(sessionId, queryResult);
  session.generator = queryResult[Symbol.asyncIterator]();
  
  try {
    // 4. Stream messages from Claude SDK
    for await (const message of queryResult) {
      const cleaned = deserializeMessage(message, generatorId);
      processMessage(cleaned, ctx, session, msgOpts);
    }
  } catch (error) {
    // Cancel handling + error classification
    if (session.cancelledByUser) {
      persistCancellation(sessionId, "claude", options.model);
      return;
    }
    notifyAndRecordError(sessionId, "claude", classifyError(error));
  } finally {
    // Cleanup: only if generator still owns session (no rapid re-query)
    const currentSession = getSession(sessionId);
    if (currentSession === session) {
      deleteQuery(sessionId);
      deleteSession(sessionId);
    }
  }
}
```

```typescript
function processMessage(
  cleanMessage: Record<string, unknown>,
  ctx: StreamContext,
  session: SessionState,
  opts: ProcessMessageOptions
) {
  // 1. Capture SDK session_id (one-shot, skip on resume)
  if (!opts.isResume && !session.agentSessionIdCaptured && cleanMessage.session_id) {
    saveAgentSessionId(sessionId, cleanMessage.session_id);
    session.agentSessionIdCaptured = true;
  }
  
  // 2. Persist assistant message to DB
  const writeResult = saveAssistantMessage(sessionId, msg, opts.model);
  
  // 3. Persist tool_result messages
  if (toolResult) {
    saveToolResultMessage(sessionId, toolResult);
  }
  
  // 4. Send to frontend (AFTER DB writes)
  FrontendClient.sendMessage({
    id: sessionId,
    type: "message",
    agentType: "claude",
    data: { type: "assistant", message: msg },
  });
  
  // 5. Classify stop_reason → error notification
  const stopError = classifyStopReason(msg.stop_reason);
  if (stopError) {
    FrontendClient.sendError(stopError);
  }
  
  // 6. Detect terminal states
  if (msg.type === "result" && msg.subtype === "success") {
    ctx.querySucceeded = true;
    updateSessionStatus(sessionId, "idle");
  }
}
```

```typescript
private async processQuery(
  sessionId: string,
  prompt: string,
  options: QueryOptions,
  existingThreadId?: string
): Promise<void> {
  const abortController = new AbortController();
  const session = { threadId: existingThreadId, abortController, isRunning: true };
  setCodexSession(sessionId, session);
  
  try {
    // Dynamic ESM import (Codex SDK is ESM-only)
    const { Codex } = await import("@openai/codex-sdk");
    
    const codex = new Codex({
      apiKey: env.OPENAI_API_KEY,
      codexPathOverride: getCodexExecutablePath(),
      env,
    });
    
    // Start or resume thread
    const thread = existingThreadId
      ? codex.resumeThread(existingThreadId, threadOptions)
      : codex.startThread(threadOptions);
    
    // Run streamed (fetch full events stream, not async iterator)
    const { events } = await thread.runStreamed(prompt, {
      signal: abortController.signal,
    });
    
    // Process event types
    for await (const event of events) {
      match(event)
        .with({ type: "thread.started" }, (e) => {
          session.threadId = e.thread_id;
        })
        .with({ type: P.union("item.started", "item.updated", "item.completed") }, (e) => {
          // Map Codex event → Claude content blocks (for frontend compatibility)
          const blocks = mapItemToContentBlocks(e.item);
          
          // Send real-time update
          FrontendClient.sendMessage({
            id: sessionId,
            type: "message",
            agentType: "codex",
            data: { type: "assistant", message: { content: blocks } },
          });
          
          // Persist completed items
          if (e.type === "item.completed") {
            saveAssistantMessage(sessionId, { content: blocks }, resolveCodexModel(options.model));
          }
        })
        .with({ type: "turn.completed" }, (e) => {
          FrontendClient.sendMessage({
            id: sessionId,
            type: "message",
            agentType: "codex",
            data: { type: "result", subtype: "success", usage: e.usage },
          });
          updateSessionStatus(sessionId, "idle");
        })
        .with({ type: "turn.failed" }, (e) => {
          const classified = classifyError(e.error);
          notifyAndRecordError(sessionId, "codex", classified);
        });
    }
  } catch (error) {
    // Same error path as Claude
    const classified = classifyError(error);
    if (classified.category === "abort") {
      persistCancellation(sessionId, "codex", model);
    } else {
      notifyAndRecordError(sessionId, "codex", classified);
    }
  } finally {
    // Cleanup session
    const currentSession = getCodexSession(sessionId);
    if (currentSession === session) {
      currentSession.isRunning = false;
    }
  }
}
```

```typescript
export function saveUserMessage(sessionId: string, content: string, model: string): WriteResult {
  const db = getDatabase();
  
  try {
    // Wrap in transaction: save message + set status='working'
    const saveAndActivate = db.transaction(() => {
      db.prepare(`
        INSERT INTO messages (id, session_id, role, content, sent_at, model)
        VALUES (?, ?, 'user', ?, ?, ?)
      `).run(messageId, sessionId, content, sentAt, model);
      
      db.prepare(`
        UPDATE sessions SET status = 'working', last_user_message_at = ?,
         error_message = NULL, error_category = NULL,
         updated_at = datetime('now') WHERE id = ?
      `).run(sentAt, sessionId);
    });
    
    saveAndActivate();
    notifyBackend(NOTIFY_SESSION_MESSAGE, sessionId); // Async notify
    return { ok: true, value: messageId };
  } catch (error) {
    return { ok: false, error: getErrorMessage(error) };
  }
}
```

```typescript
export function saveAssistantMessage(
  sessionId: string,
  message: { id?: string; role?: string; content?: unknown; stop_reason?: string },
  model: string
): WriteResult {
  // Handle cancelled messages with special envelope
  const contentPayload = message.stop_reason === "cancelled"
    ? { message: { stop_reason: "cancelled" }, blocks: message.content ?? [] }
    : (message.content ?? []);
  
  db.prepare(`
    INSERT INTO messages (id, session_id, role, content, sent_at, model, agent_message_id)
    VALUES (?, ?, 'assistant', ?, ?, ?, ?)
  `).run(messageId, sessionId, JSON.stringify(contentPayload), sentAt, model, message.id || null);
  
  notifyBackend(NOTIFY_SESSION_MESSAGE, sessionId);
}
```
